

using SQLite

Set mode to tabs for all imports:

.mode tabs

----------------------------------------------------------------
create table kanji (
    kanji         TEXT PRIMARY KEY, 
    kanji_damage_url    	  TEXT NOT NULL
);

create table kanji_component (
    id			  INTEGER PRIMARY KEY, 
    kanji_id         TEXT NOT NULL, 
	component_id     TEXT NOT NULL,
	ranking_most_used	INTEGER,
	ranking				INTEGER NOT NULL,
);

# don't need to set autoincrement, INTEGER PRIMAY KEY is autoincremented whenever not provided.
